 function PostsCtrlAjax($scope, $http)
        {
            $http({method: 'POST', url: 'client/menu.json'}).success(function(data)
            {
                $scope.posts = data; // response data
            });
        }
        
        function Combo($scope, $http)
        {
            $http({method: 'POST', url: 'client/combos.json'}).success(function(data)
            {
                $scope.comboboxes = data; // response data
            });
        }
